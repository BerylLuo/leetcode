62、Unique Paths
```
func uniquePaths(m int, n int) int {
    arr := make([][]int, m)
    for i := 0; i < m; i++ {
        arr[i] = make([]int, n)
    }
    arr[0][0] = 1
    for i := 0; i < m; i++ {
        for j := 0; j < n; j++ {
            if i > 0 {
                arr[i][j] += arr[i-1][j]
            }
            if j > 0 {
                arr[i][j] += arr[i][j-1]
            }
        }
    }
    return arr[m-1][n-1]
}
```
