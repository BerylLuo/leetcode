func uniquePaths(m int, n int) int {
    arr := make([][]int, m)
    for i := 0; i < m; i++ {
        arr[i] = make([]int, n)
    }
    arr[0][0] = 1
    for i := 0; i < m; i++ {
        for j := 0; j < n; j++ {
            if i > 0 {
                arr[i][j] += arr[i-1][j]
            }
            if j > 0 {
                arr[i][j] += arr[i][j-1]
            }
        }
    }
    return arr[m-1][n-1]
}

func uniquePaths(m int, n int) int {
    var mydict = map[[2]int]int{
        [2]int{0, 0}: 1,
    }
    return helper(m-1, n-1, mydict)
}
func helper(m int, n int, mydict map[[2]int]int) int {
    if v, ok := mydict[[2]int{m, n}]; ok {
        return v
    }
    var v int
    if m > 0 {
        v += helper(m-1, n, mydict)
    }
    if n > 0 {
        v += helper(m, n-1, mydict)
    }
    mydict[[2]int{m, n}] = v
    return v
}
