var d = map[byte]string{
	'2': "abc",
	'3': "def",
	'4': "ghi",
	'5': "jkl",
	'6': "mno",
	'7': "pqrs",
	'8': "tuv",
	'9': "wxyz",
}

func letterCombinations(digits string) []string {
	var res []string
    if len(digits) == 0 {
        return res
    }
	if len(digits) == 1 {
		return strings.Split(d[digits[0]], "")
	} else {
		subres := letterCombinations(digits[1:])
		for _, r := range d[digits[0]] {
			for _, s := range subres {
				res = append(res, string(r) + s)
			}
		}
	}
	return res
}
